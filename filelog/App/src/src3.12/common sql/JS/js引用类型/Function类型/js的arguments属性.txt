<script>
function jiechen(num)
{
  if(num==1)
 {return num}
else
{
  return num*jiechen(num-1);
}
}

var jiechen1=jiechen;
jiechen=function(){return 10;}

var n=jiechen(5);
var nn=jiechen1(5);//返回50 我们可以看到在递归内部调用jiechen时已经不是原来的函数而是返回5*10


function jiechenArgument(num)
{
  if(num==1)
 {return num}
else
{
  return num*arguments.callee(num-1);
}
}

var jiechen11=jiechenArgument;
jiechenArgument=function(){return 10;}

var nnn=jiechenArgument(5);
var nnnn=jiechen11(5);//120使用arguments.calle属性此属性指向当前函数 避免上面与函数名的耦合。


document.write(n);
document.write('<br/>');
document.write(nn);
document.write('<br/>');
document.write(nnn);
document.write('<br/>');
document.write(nnnn);
</script>
